3. Code Program perhitungan manual (menggunakan dimensi 10x10) histogram dan ekualisasi histogram.

import matplotlib.pyplot as plt
import numpy as np
# 3. Histogram and Histogram Equalization:

# a) Histogram Calculation (manual for 10x10)
# Create a histogram (frequency of each gray level)
histogram = [0] * 256  # Assuming 8-bit grayscale image
for row in img_resized3:
    for pixel in row:
        histogram[pixel] += 1

# Display the histogram
plt.figure(figsize=(8, 4))
plt.subplot(1, 2, 1)
plt.bar(range(256), histogram)
plt.title('Histogram')
plt.xlabel('Gray Level')
plt.ylabel('Frequency')

# b) Histogram Equalization (manual for 10x10)
# Calculate the cumulative distribution function (CDF)
total_pixels = img_resized3.size
cdf = [0] * 256
for i in range(256):
    cdf[i] = sum(histogram[:i+1]) / total_pixels

# Apply the CDF to the image to equalize it
equalized_image = np.zeros_like(img_resized3)
for i in range(img_resized3.shape[0]):
    for j in range(img_resized3.shape[1]):
        equalized_image[i, j] = round(cdf[img_resized3[i, j]] * 255)

# Calculate the histogram of the equalized image
equalized_histogram = [0] * 256
for row in equalized_image:
    for pixel in row:
        equalized_histogram[int(pixel)] += 1

# Display the histogram of the equalized image
plt.subplot(1, 2, 2)
plt.bar(range(256), equalized_histogram)
plt.title('Equalized Histogram')
plt.xlabel('Gray Level')
plt.ylabel('Frequency')

plt.show()
